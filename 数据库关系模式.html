数据库关系模式（CogniFlow） ───────────────────────────────────────────────── users (id, username, password, email, role) courses (id, name, description, category) knowledge_points (id, name, description, difficulty, order_index) FK: course_id → courses(id) FK: parent_id → knowledge_points(id) questions (id, type, title, option_a, option_b, option_c, option_d, answer, analysis, score, difficulty) FK: course_id → courses(id) FK: knowledge_point_id → knowledge_points(id) study_records (id, status, score, time_spent) FK: user_id → users(id) FK: knowledge_point_id → knowledge_points(id) study_plans (id, title, plan_type, content, total_days, progress_days) FK: user_id → users(id) chat_messages (id, role, content) FK: user_id → users(id) exam_records (id, total_score, user_score, correct_count, question_count, question_ids, status, start_time, end_time) FK: user_id → users(id) FK: course_id → courses(id) exam_answers (id, user_answer, is_correct, score) FK: exam_record_id → exam_records(id) FK: question_id → questions(id) announcements (id, title, content, pinned) FK: publisher_id → users(id) course_requests (id, topic, description, status) FK: user_id → users(id) system_logs (id, action, detail) FK: user_id → users(id)
